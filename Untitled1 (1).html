import os
from rdkit import Chem
from rdkit.Chem import AllChem

# Initialize the list to store converted ligands
converted = []

# Convert and save the first 50 filtered molecules to PDBQT format
for idx, mol in enumerate(filtered_molecules[:50]):
    # Generate 3D coordinates
    mol = Chem.AddHs(mol)
    AllChem.EmbedMolecule(mol, AllChem.ETKDG())
    
    # Convert the molecule to PDBQT format and store in the 'converted' list
    pdbqt_content = Chem.MolToPDBBlock(mol)
    converted.append(pdbqt_content)
    
    print(f"Converted ligand {idx+1} to PDBQT format")

# Define the directory where you want to save PDBQT files
output_dir = r"C:\PSO\PDBQT_files"

# Create the output directory if it doesn't exist
if not os.path.exists(output_dir):
    os.makedirs(output_dir)

# Save the converted molecules to PDBQT files
for idx, pdbqt_content in enumerate(converted):
    # Define the output PDBQT file path
    output_path = os.path.join(output_dir, f"ligand_{idx+1}.pdbqt")
    
    # Save the PDBQT file
    with open(output_path, 'w') as f:
        f.write(pdbqt_content)
    
    print(f"Saved converted ligand {idx+1} to PDBQT file: {output_path}")

import os

# AutoDock Vina executable path
vina_path = r"C:\Program Files (x86)\The Scripps Research Institute\Vina\vina.exe"

# Ligand PDBQT directory
ligand_dir = r"C:\PSO\PDBQT_files"

# Protein PDB path
protein_pdb_path = r"C:\PSO\8gcy.pdb"

# Convert protein to PDBQT format
protein_pdbqt_path = os.path.join(ligand_dir, "protein.pdbqt")
os.system(f'"{vina_path}" --receptor "{protein_pdb_path}" --ligand "{protein_pdb_path}" --out "{protein_pdbqt_path}" --center_x 0 --center_y 0 --center_z 0 --size_x 20 --size_y 20 --size_z 20')

# Docking for the first 3 ligands
for idx in range(1, 4):
    ligand_path = os.path.join(ligand_dir, f"ligand_{idx}.pdbqt")

    # Run the docking command and capture the result
    docking_result = os.popen(docking_command).read()
    
    # Print the docking result for visualization
    print(f"Docking result for ligand {idx}:\n{docking_result}")
